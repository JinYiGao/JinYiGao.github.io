1# 当纹理坐标设置在(0,0) (1,1)之外时，纹理重复方式

GL_REPEAT 对纹理的默认行为。重复纹理图像。

7.1 材质

 1//物体材质信息
 2struct Material{
 3    vec3 ambient;//环境光照下物体反射的颜色，通常和物体本身颜色一致
 4    vec3 diffuse;//漫反射光照下物体的颜色，通常也为物体本身颜色
 5    vec3 specular;//镜面光照对物体的颜色影响，类似于specularstrength
 6    float shininess;//反光度
 7};
 8uniform Material material;
 9    
10struct Light{
11    vec3 position;
12
13    vec3 ambient;//环境光
14    vec3 diffuse;//漫反射光
15    vec3 specular;//镜面反射光
16};
17uniform Light light;
18
19void main()
20{
21    //环境光照
22    vec3 ambient = material.ambient * light.ambient;
23
24    //漫反射光照计算 在世界坐标下进行
25    vec3 norm = normalize(Normal);
26    vec3 lightDir = normalize(lightPos - FragPos);//光线方向
27    float diff = max(dot(norm, lightDir), 0.0);
28    vec3 diffuse = (diff * material.diffuse) * light.diffuse;
29
30    //镜面反射 在观察空间下进行
31    vec3 viewDir = normalize(-vec3(view * vec4(FragPos, 1.0)));
32    vec3 reflectDir = reflect(-vec3(normalize(view*vec4(lightPos,1.0) - 	view*vec4(FragPos,1.0))), mat3(transpose(inverse(view))) * Normal);
33    float spec = pow(max(dot(viewDir, reflectDir), 0.0), 					material.shininess);
34    vec3 specular = (material.specular * spec) * light.specular;
35
36    vec3 result = (ambient + diffuse + specular);
37    FragColor = vec4(result, 1.0);
38}

 4
 5//清除颜色缓冲和深度缓冲
 6glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
 7
 8//禁用深度缓冲写入 即 使用一个只读的深度缓冲 
 9//***只在深度测试被启用的时候才有效果
10glDepthMask(GL_FALSE);
11
12//深度测试函数 
13//即 通过修改比较运算符, 让我们来决定什么时候该通过或者丢弃一个片段
14glDepthFunc(GL_LESS);
函数 描述
GL_ALWAYS 永远通过深度测试
GL_NEVER 永远不通过深度测试

GL_LEQUAL 在片段深度值小于等于缓冲区的深度值时通过测试
GL_GREATER 在片段深度值大于缓冲区的深度值时通过测试
GL_NOTEQUAL 在片段深度值不等于缓冲区的深度值时通过测试
GL_GEQUAL 在片段深度值大于等于缓冲区的深度值时通过测试

9.2 模板测试

 1//其本质原理类比于模板操作
 2//通过设置一个viewport的模板值，即模板缓冲，基于一定的规则 来决定缓冲对应的片段是否应该被绘制.以及模板测试时缓冲的内容应该如何更新.
 3
 4//模板缓冲操作允许我们在渲染片段时将模板缓冲设定为一个特定的值。通过在渲染时修改模板缓冲的内容，我们写入了模板缓冲。在同一个（或者接下来的）渲染迭代中，我们可以读取这些值，来决定丢弃还是保留某个片段。
 5
 6//启用模板测试
 7glEnable(GL_STENCIL_TEST);
 8
 9//每次渲染迭代前 清除模板缓冲
10glClear(GL_STENCIL_BUFFER_BIT);
11
12glStencilMask(0xFF); // 每一位写入模板缓冲时都保持原样(正常写入)
13glStencilMask(0x00); // 每一位在写入模板缓冲时都会变成0(禁用写入)
14
15//--------------------模板函数------------------------
16// ---该函数决定了以何种规则进行比较，以确定模板测试是否通过
17glStencilFunc(GLenum func, GLint ref, GLuint mask);
18// 1.func：设置模板测试函数(Stencil Test Function)。这个测试函数将会应用到已储存的模板值上和glStencilFunc函数的ref值上。可用的选项有：GL_NEVER、GL_LESS、GL_LEQUAL、GL_GREATER、GL_GEQUAL、GL_EQUAL、GL_NOTEQUAL和GL_ALWAYS。它们的语义和深度缓冲的函数类似。
19// 2.ref：设置了模板测试的参考值(Reference Value)。模板缓冲的内容将会与这个值进行比较。
20// 3.mask：设置一个掩码，它将会与参考值和储存的模板值在测试比较它们之前进行与(AND)运算。初始情况下所有位都为1
21
22// ---该函数决定了模板测试通过或失败时该如何更新模板缓冲值 
23glStencilOp(GLenum sfail, GLenum dpfail, GLenum dppass);
24// 1.sfail：模板测试失败时采取的行为。
25// 2.dpfail：模板测试通过，但深度测试失败时采取的行为。
26// 3.dppass：模板测试和深度测试都通过时采取的行为。
27// *行为及其描述见下表:
28// #默认情况下glStencilOp是设置为(GL_KEEP, GL_KEEP, GL_KEEP)的，所以不论任何测试的结果是如何，模板缓冲都会保留它的值
行为 描述
GL_KEEP 保持当前储存的模板值
GL_ZERO 将模板值设置为0
GL_REPLACE 将模板值设置为glStencilFunc函数设置的ref值
GL_INCR 如果模板值小于最大值则将模板值加1
GL_INCR_WRAP 与GL_INCR一样，但如果模板值超过了最大值则归零
GL_DECR 如果模板值大于最小值则将模板值减1
GL_DECR_WRAP 与GL_DECR一样，但如果模板值小于0则将其设置为最大值
GL_INVERT 按位翻转当前的模板缓冲值

4
5//混合函数 ---设置源和目标因子值

选项 值
GL_ZERO 因子等于0
GL_ONE 因子等于1
GL_SRC_COLOR 因子等于源颜色向量C¯source
GL_ONE_MINUS_SRC_COLOR 因子等于1−C¯source

GL_ONE_MINUS_DST_COLOR 因子等于1−C¯destination
GL_SRC_ALPHA 因子等于C¯sourceC¯source的alphaalpha分量
GL_ONE_MINUS_SRC_ALPHA 因子等于1−1− C¯sourceC¯source的alphaalpha分量
GL_DST_ALPHA 因子等于C¯destinationC¯destination的alphaalpha分量

9.8.3 Uniform缓冲对象(Uniform块)

 1//=============================Uniform缓冲对象============================
 2//------------------------------uniform Block块布局----------------------------
 3//***Uniform块的内容是储存在一个缓冲对象中的，它实际上只是一块预留内存。因为这块内存并不会保存它具体保存的是什么类型的数据，我们还需要告诉OpenGL内存的哪一部分对应着着色器中的哪一个uniform变量***
 4//---Uniform块---
 5layout (std140) uniform ExampleBlock
 6{
 7    float value;
 8    vec3  vector;
 9    mat4  matrix;
10    float values[3];
11    bool  boolean;
12    int   integer;
13};
14
15//每个变量都有一个基准对齐量(Base Alignment)，它等于一个变量在Uniform块中所占据的空间（包括填充量(Padding)），这个基准对齐量是使用std140布局的规则计算出来的。接下来，对每个变量，我们再计算它的对齐偏移量(Aligned Offset)，它是一个变量从块起始位置的字节偏移量。一个变量的对齐字节偏移量必须等于基准对齐量的倍数。
16layout (std140) uniform ExampleBlock
17{
18                     // 基准对齐量       // 对齐偏移量
19    float value;     // 4               // 0 
20    vec3 vector;     // 16              // 16  (必须是16的倍数，所以 4->16)
21    mat4 matrix;     // 16              // 32  (列 0)
22                     // 16              // 48  (列 1)
23                     // 16              // 64  (列 2)
24                     // 16              // 80  (列 3)
25    float values[3]; // 16              // 96  (values[0])
26                     // 16              // 112 (values[1])
27                     // 16              // 128 (values[2])
28    bool boolean;    // 4               // 144
29    int integer;     // 4               // 148
30}; 
31
32//-----------------------------使用Uniform缓冲----------------------------
33// * 首先跟其他缓冲一样必须在程序中创建
34unsigned int uboExampleBlock;
35glGenBuffers(1, &uboExampleBlock);
36glBindBuffer(GL_UNIFORM_BUFFER, uboExampleBlock);
37glBufferData(GL_UNIFORM_BUFFER, 152, NULL, GL_STATIC_DRAW); // 分配152字节的内存
38
39//***如何才能让OpenGL知道哪个Uniform缓冲对应的是哪个Uniform块呢？***
40//在OpenGL上下文中，定义了一些绑定点(Binding Point)，我们可以将一个Uniform缓冲链接至它。在创建Uniform缓冲之后，我们将它绑定到其中一个绑定点上，并将着色器中的Uniform块绑定到相同的绑定点，把它们连接到一起。
41
42//将Uniform块绑定到一个特定的绑定点
43//先查询着色器中Uniform块索引，绑定到绑定点2
44unsigned int lights_index = glGetUniformBlockIndex(shaderA.ID,"Lights"); 
45glUniformBlockBinding(shaderA.ID, lights_index, 2);
46//再将Uniform缓冲绑定到相同的绑定点2
47glBindBufferBase(GL_UNIFORM_BUFFER, 2, uboExampleBlock); 
48// 或
49glBindBufferRange(GL_UNIFORM_BUFFER, 2, uboExampleBlock, 0, 152);
50
51//***从OpenGL 4.2版本起，也可以添加一个布局标识符，显式地将Uniform块的绑定点储存在着色器中，这样就不用再调用glGetUniformBlockIndex和glUniformBlockBinding了。***
52// 例如:
53layout(std140, binding = 2) uniform Lights { ... };
54    
55//填充或者更新Uniform缓冲
56glBindBuffer(GL_UNIFORM_BUFFER, uboExampleBlock);
57int b = true; // GLSL中的bool是4字节的，所以我们将它存为一个integer
58glBufferSubData(GL_UNIFORM_BUFFER, 144, 4, &b); 
59glBindBuffer(GL_UNIFORM_BUFFER, 0);
